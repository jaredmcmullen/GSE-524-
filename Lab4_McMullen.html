menuism <- function(url) {
  link <-read_html(url)
  html <- html_nodes(link, css=".list-unstyled-links a")
  readtext <-html_text(html)[1:51]
  name <- html_nodes(link, css=".restaurant-title span")
  readname <- html_text(name)
  thestate = str_extract(readtext, "(\\w+)")
  readtext <-readtext %>% 
    str_replace("locations", "") %>% 
    str_replace_all("[()]", "")
  df <- data.frame(
    Company = readname,
    StateABB = stateabb(thestate),
    Count = as.numeric(str_extract(readtext, "[:digit:]+")))
  df<-na.omit(df)
  view(df)
}
