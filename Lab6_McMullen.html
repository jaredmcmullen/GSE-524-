datatable(insurance2)
model1 <- lr_model %>% 
  fit(charges ~ age, data = insurance)
model2 <- lr_model %>% 
  fit(charges ~ age + bmi, data = insurance)
model3 <- lr_model %>% 
  fit(charges ~ age + bmi + smoker, data = insurance)
model4 <- lr_model %>% 
  fit(charges ~ (age + bmi):smoker, data = insurance)
model5 <- lr_model %>% 
  fit(charges ~ (age + bmi)*smoker, data = insurance)
insurance2<- insurance2 %>% 
  mutate(
    Pred1 = predict(model1, 
                    new_data = insurance2,
                    type = "raw"),
    Pred2 = predict(model2, 
                    new_data = insurance2,
                    type = "raw"),
    Pred3 = predict(model3, 
                    new_data = insurance2,
                    type = "raw"),
    Pred4 = predict(model4, 
                    new_data = insurance2,
                    type = "raw"),
    Pred5 = predict(model5, 
                    new_data = insurance2,
                    type = "raw"),
    MSE1 = (Pred1 - charges)^2,
    MSE2 = (Pred2 - charges)^2,
    MSE3 = (Pred3 - charges)^2,
    MSE4 = (Pred4 - charges)^2,
    MSE5 = (Pred5 - charges)^2
  )

MSE1 <- round(mean(insurance2$MSE1), digits = 0)
MSE2 <- round(mean(insurance2$MSE2), digits = 0)
MSE3 <- round(mean(insurance2$MSE3), digits = 0)
MSE4 <- round(mean(insurance2$MSE4), digits = 0)
MSE5 <- round(mean(insurance2$MSE5), digits = 0)

df <- data.frame(
  Model_num = c("Model1", "Model2", "Model3", "Model4", "Model5"),
  MSE = c(MSE1, MSE2, MSE3, MSE4, MSE5)
)

datatable(df)

From the table, we can determine that the MSE that is the lowest is for model5. It seems that as the model number goes up, the MSE value drops. If the best model to use is model 5, then when both age and BMI come from a smoker or a non smoker, a better prediction can be made about the charges someone will receive.
